76// Get Firwmware App Version - version is placed in General Purpose Read Registers as follows:
77// GPR_READ04 - Version (Major)
78// GPR_READ05 - Version (Minor)

88#define SFE_ENS160_TEMP_IN 0x13
89// Temperature compensation can be given to the sensor for more accurate
90// readings.
91// Temperature should be given in Kelvin in the following format:
92// Value: Temperature in Kelvin * 64
93// Converting Celsius to Kelvin = Temp + 273.15
94// Ergo: (Temp + 273.15) * 64
95typedef struct
96{
97 uint8_t temp_in_msb : 8; // 0x14 -- Integer Part
98 uint8_t temp_in_lsb : 8; // 0x13 -- Fractions Part
99} sfe_ens160_temp_in_t;
100
101#define SFE_ENS160_RH_IN 0x15
102// Relative Humidity compensation can be given to the sensor for more accurate
103// readings.
104// RH should be given in %rH * 512
105typedef struct
106{
107 uint8_t rh_in_msb : 8; // 0x16 -- Integer Part
108 uint8_t rh_in_lsb : 8; // 0x15 -- Fraction Part
109} sfe_ens160_rh_in_t;
110

119 uint8_t new_dat : 1;
